Step 3 — Service Worker

⚙️ sw.js — Service Worker

Create this file at /public/sw.js. It handles offline caching and background push notifications. Then uncomment the registration line in both HTML files.

const CACHE_NAME = 'arivo-v1';
const STATIC_ASSETS = [
  '/',
  '/guest/',
  '/dashboard/',
  '/icons/icon-192.png',
  '/icons/icon-512.png'
];

// Install — cache static assets
self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME).then(cache => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

// Activate — clean old caches
self.addEventListener('activate', event => {
  event.waitUntil(
    caches.keys().then(keys =>
      Promise.all(keys.filter(k => k !== CACHE_NAME).map(k => caches.delete(k)))
    )
  );
  self.clients.claim();
});

// Fetch — serve from cache, fall back to network
self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request).then(cached => cached || fetch(event.request))
  );
});

// Push — show notification when server sends a push
self.addEventListener('push', event => {
  const data = event.data ? event.data.json() : {};
  const title = data.title || 'Luvia';
  const options = {
    body: data.body || 'New notification',
    icon: '/icons/icon-192.png',
    badge: '/icons/icon-192.png',
    tag: data.tag || 'arivo-notif',
    data: { url: data.url || '/dashboard/' },
    requireInteraction: data.urgent || false
  };
  event.waitUntil(self.registration.showNotification(title, options));
});

// Notification click — open the dashboard
self.addEventListener('notificationclick', event => {
  event.notification.close();
  event.waitUntil(
    clients.openWindow(event.notification.data.url || '/dashboard/')
  );
});

Step 4 — Server-side Push Notifications

🔔 Web Push setup (Node.js / Supabase Edge Function)

Use the web-push npm package to send push notifications from your server when a guest submits a request. This is what actually reaches the staff device even when the browser is closed.

# Install
npm install web-push

# Generate VAPID keys (run once, store securely)
npx web-push generate-vapid-keys
// server/push.js
const webPush = require('web-push');

webPush.setVapidDetails(
  'mailto:[email protected]',
  process.env.VAPID_PUBLIC_KEY,
  process.env.VAPID_PRIVATE_KEY
);

// Call this when a guest submits a request
async function notifyStaff(subscription, requestTitle, roomNumber) {
  const payload = JSON.stringify({
    title: '🛎 New Guest Request',
    body: `${requestTitle} — Room ${roomNumber}`,
    tag: 'req-' + Date.now(),
    url: '/dashboard/requests'
  });
  try {
    await webPush.sendNotification(subscription, payload);
  } catch (err) {
    console.error('Push failed:', err);
    // If subscription expired, remove it from database
    if (err.statusCode === 410) {
      await db.deleteSubscription(subscription.endpoint);
    }
  }
}

The subscription object is generated in the browser when the manager clicks "Enable notifications" and must be stored in your database (Supabase) per hotel. When any guest at that hotel submits a request, look up the hotel's subscription and send the push.

Step 5 — Saving Push Subscriptions

💾 Store subscription in Supabase

In the dashboard JS, after the manager clicks Enable, save their subscription to the database so the server can push to them even when the tab is closed.

// In dashboard HTML — update requestNotifPermission()
async function requestNotifPermission() {
  const perm = await Notification.requestPermission();
  if (perm !== 'granted') return;

  // Register service worker first
  const reg = await navigator.serviceWorker.ready;

  // Subscribe to push
  const subscription = await reg.pushManager.subscribe({
    userVisibleOnly: true,
    applicationServerKey: urlBase64ToUint8Array(VAPID_PUBLIC_KEY)
  });

  // Save to Supabase
  await fetch('/api/push-subscription', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      hotelId: HOTEL_ID,
      subscription: subscription.toJSON()
    })
  });

  notifGranted = true;
  toast('🔔 Push notifications enabled');
}

// Helper
function urlBase64ToUint8Array(base64String) {
  const padding = '='.repeat((4 - base64String.length % 4) % 4);
  const base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/');
  const rawData = atob(base64);
  return new Uint8Array([...rawData].map(c => c.charCodeAt(0)));
}

Important notes for engineers

🔒 HTTPS required

PWA features — including service workers, push notifications, and the install prompt — only work over HTTPS. Localhost is the only exception during development. Make sure your production domain has a valid SSL certificate (Cloudflare or Let's Encrypt are both free).

📱 "Add to Home Screen" instructions for hotels

Hotels need to know how to tell their staff to install the dashboard as an app. Include this in your onboarding email:

For iPhone (Safari only):

1. Open luviaguest.com in Safari

2. Tap the Share button (box with arrow)

3. Tap "Add to Home Screen" → "Add"

For Android (Chrome):

1. Open luviaguest.com in Chrome

2. Tap the banner "Add Luvia to Home Screen" or

3. Tap ⋮ menu → "Add to Home Screen"
